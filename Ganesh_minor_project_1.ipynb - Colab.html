# ============================================================
# GROUPDNA - WhatsApp Group Chat Analyzer
# Built using only Python fundamentals + NumPy
# No pandas, no matplotlib, no regex, no collections.Counter
# ============================================================
# ============================================================
# FEATURE 1 - THE CHAT PARSER
# ============================================================
import numpy as np
from datetime import datetime
# open the file and read all lines
with open('hostel_bois.txt', 'r', encoding='utf-8') as f:
    lines = f.read().split('\n')
# counters for skipped lines
system_count = 0
media_count = 0
deleted_count = 0
# this list will hold all real messages as dictionaries
messages = []
for line in lines:
    # skip empty lines
    if line.strip() == '':
        continue
    # check if line starts with a date (DD/MM/YY pattern)
    # first 8 chars should look like "01/04/24"
    if len(line) < 8:
        continue
    first8 = line[:8]
    # check if position 2 and 5 are slashes (date format check)
    if first8[2] != '/' or first8[5] != '/':
        continue
    # try to split into timestamp and the rest
    # format: "DD/MM/YY, HH:MM - Sender: message"
    if ' - ' not in line:
        continue
    # split at first ' - ' to get timestamp and the rest
    parts = line.split(' - ', 1)
    timestamp_str = parts[0].strip()   # "01/04/24, 01:13"
    rest = parts[1].strip()            # "Rahul: scene fix"
    # check if there is a colon in rest (to identify real messages)
    # system messages like "Priya created group" have no ": "
    if ': ' not in rest:
        system_count += 1
        continue
    # split at first ': ' to get sender and message
    sender_parts = rest.split(': ', 1)
    sender = sender_parts[0].strip()
    message = sender_parts[1].strip()
    # handle media omitted messages
    if message == '<Media omitted>':
        media_count += 1
        # still count as a message from sender but mark it
        messages.append({
            'timestamp': timestamp_str,
            'sender': sender,
            'text': message,
            'is_media': True,
            'is_deleted': False
        })
        continue
    # handle deleted messages
    if message == 'This message was deleted':
        deleted_count += 1
        messages.append({
            'timestamp': timestamp_str,
            'sender': sender,
            'text': message,
            'is_media': False,
            'is_deleted': True
        })
        continue
    # normal real message
    messages.append({
        'timestamp': timestamp_str,
        'sender': sender,
        'text': message,
        'is_media': False,
        'is_deleted': False
    })
# get list of unique participants using a set
participants = list(set(m['sender'] for m in messages))
participants.sort()
print(f"Successfully parsed {len(messages)} messages from {len(participants)} participants.")
print(f"Skipped: {system_count} system messages, {media_count} media-omitted, {deleted_count} deleted messages.")
print(f"First message: {messages[0]}")
print(f"Last message:  {messages[-1]}")
# ============================================================
# FEATURE 2 - GROUP OVERVIEW
# ============================================================
# count messages per person using a plain dict
msg_count = {}
for m in messages:
    name = m['sender']
    if name not in msg_count:
        msg_count[name] = 0
    msg_count[name] += 1
# sort by message count, highest first
sorted_members = sorted(msg_count.items(), key=lambda x: x[1], reverse=True)
total_messages = len(messages)
# get date range from first and last message timestamps
first_date_str = messages[0]['timestamp']   # "01/04/24, 01:12"
last_date_str  = messages[-1]['timestamp']
# parse just the date part to show in report
first_dt = datetime.strptime(first_date_str, '%d/%m/%y, %H:%M')
last_dt  = datetime.strptime(last_date_str,  '%d/%m/%y, %H:%M')
# number of days the group was active
total_days = (last_dt - first_dt).days + 1
print()
print('=' * 60)
print('       GROUPDNA REPORT — "Hostel Bois 4ever"')
print(f'       {total_days} days  •  {total_messages:,} messages  •  {len(participants)} members')
print('=' * 60)
print(f'  Period       : {first_dt.strftime("%d %B %Y")} to {last_dt.strftime("%d %B %Y")}')
print()
print('  MESSAGES PER PERSON')
# max messages for bar scaling
max_msgs = sorted_members[0][1]
for name, count in sorted_members:
    pct = (count / total_messages) * 100
    # bar length proportional to message count (max 20 blocks)
    bar_len = int((count / max_msgs) * 20)
    if bar_len == 0:
        bar = '.'
    else:
        bar = '█' * bar_len
    print(f'  {name:<8} {bar:<22} {count:>4} ({pct:>4.1f}%)')
# ============================================================
# FEATURE 3 - MOST ACTIVE DAY AND HOUR
# ============================================================
# count messages by date (string key like "01/04/24")
day_count = {}
hour_count = {}
for m in messages:
    ts = m['timestamp']   # "01/04/24, 23:14"
    date_part = ts.split(',')[0].strip()   # "01/04/24"
    hour_part = int(ts.split(',')[1].strip().split(':')[0])  # 23
    # increment day count
    if date_part not in day_count:
        day_count[date_part] = 0
    day_count[date_part] += 1
    # increment hour count
    if hour_part not in hour_count:
        hour_count[hour_part] = 0
    hour_count[hour_part] += 1
# find the busiest day using max with key
busiest_day_str = max(day_count, key=lambda x: day_count[x])
busiest_day_count = day_count[busiest_day_str]
# convert "14/04/24" to a readable date
busiest_day_dt = datetime.strptime(busiest_day_str, '%d/%m/%y')
busiest_day_readable = busiest_day_dt.strftime('%d %B %Y')
# find the busiest hour
busiest_hour = max(hour_count, key=lambda x: hour_count[x])
# average messages per day at that hour (over 60 days)
avg_at_hour = round(hour_count[busiest_hour] / total_days, 1)
print()
print(f'  Busiest day  : {busiest_day_readable} ({busiest_day_count} messages)')
print(f'  Busiest hour : {busiest_hour:02d}:00 - {busiest_hour+1:02d}:00 (avg {avg_at_hour} messages per day)')
# ============================================================
# FEATURE 4 - ACTIVITY HEATMAP (NumPy)
# ============================================================
# map each participant name to a row index
name_to_idx = {}
for i, name in enumerate(sorted_members):
    name_to_idx[name[0]] = i
num_people = len(sorted_members)
# create a 6x24 NumPy matrix filled with zeros
# rows = participants, columns = hours 0 to 23
heatmap = np.zeros((num_people, 24), dtype=int)
# fill the matrix by going through every message
for m in messages:
    sender = m['sender']
    ts = m['timestamp']
    hour = int(ts.split(',')[1].strip().split(':')[0])
    row = name_to_idx[sender]
    heatmap[row][hour] += 1   # increment the right cell
# now render the heatmap using block characters
# shading levels based on fraction of person's max activity
shading = ['. ', '░ ', '▒ ', '█ ']
print()
print('  ACTIVITY HEATMAP (messages by hour)')
print('           00  03  06  09  12  15  18  21')
for i, (name, _) in enumerate(sorted_members):
    row = heatmap[i]
    row_max = row.max()   # person's busiest hour count
    # build the visual bar showing every 3rd hour (8 columns)
    bar = ''
    for h in [0, 3, 6, 9, 12, 15, 18, 21]:
        val = row[h]
        if row_max == 0:
            level = 0
        else:
            frac = val / row_max
            if frac == 0:
                level = 0
            elif frac <= 0.25:
                level = 1
            elif frac <= 0.50:
                level = 2
            elif frac <= 0.75:
                level = 3
            else:
                level = 3
            # more precise 4-level split
            if frac == 0:
                level = 0
            elif frac < 0.25:
                level = 0
            elif frac < 0.50:
                level = 1
            elif frac < 0.75:
                level = 2
            else:
                level = 3
        bar += shading[level]
    print(f'  {name:<8} {bar}')
# ============================================================
# FEATURE 5 - TOP WORDS
# ============================================================
# stop words to skip (very common english + hinglish fillers)
stop_words = {
    'i', 'is', 'the', 'a', 'and', 'or', 'to', 'of', 'in',
    'on', 'for', 'it', 'this', 'that', 'was', 'are', 'be',
    'with', 'as', 'at', 'by', 'an', 'we', 'he', 'she', 'they',
    'you', 'me', 'my', 'so', 'if', 'but', 'not', 'have', 'has',
    'had', 'do', 'did', 'from', 'will', 'would', 'can', 'just',
    'about', 'up', 'out', 'been', 'no', 'its', 'into', 'more',
    'also', 'all', 'what', 'which', 'there', 'their', 'when',
    'who', 'how', 'one', 'then', 'than', 'him', 'her', 'his',
    'our', 'your', 'even', 'like', 'get', 'got', 'go', 'came',
    'come', 'too', 'very', 'some', 'any', 'only', 'should',
    'could', 'most', 'over', 'after', 'now', 'every', 'think',
    'know', 'want', 'see', 'said', 'each', 'way', 'them'
}
# punctuation characters to strip from words
punctuation = '.,!?;:"\'-()[]{}'
# count all words across all real messages
word_freq = {}
for m in messages:
    # skip media and deleted messages for word counting
    if m['is_media'] or m['is_deleted']:
        continue
    # tokenize: lowercase and split by spaces
    words = m['text'].lower().split()
    for word in words:
        # strip punctuation from both ends
        word = word.strip(punctuation)
        # skip empty strings, stop words, and very short words
        if len(word) < 2:
            continue
        if word in stop_words:
            continue
        # count the word
        if word not in word_freq:
            word_freq[word] = 0
        word_freq[word] += 1
# sort words by count and get top 10
top_words = sorted(word_freq.items(), key=lambda x: x[1], reverse=True)[:10]
# find max count for bar scaling
max_word_count = top_words[0][1]
print()
print("  THIS GROUP'S FAVOURITE WORDS")
for word, count in top_words:
    # bar length proportional to count (max 20 blocks)
    bar_len = int((count / max_word_count) * 20)
    bar = '█' * bar_len
    print(f'  {word:<12} {bar:<22} {count}')
# ============================================================
# FEATURE 6 - RESPONSE SPEED & SILENT STREAKS
# ============================================================
# --- Part A: Average Response Time ---
# for each message, if someone else sent the previous message,
# calculate the gap and add it to that sender's response time list
response_gaps = {}   # { sender: [gap_in_seconds, ...] }
for name in msg_count:
    response_gaps[name] = []
prev_sender = None
prev_time = None
for m in messages:
    if m['is_media'] or m['is_deleted']:
        continue
    curr_sender = m['sender']
    curr_time = datetime.strptime(m['timestamp'], '%d/%m/%y, %H:%M')
    # only count gap if the previous message was from someone else
    if prev_sender is not None and prev_sender != curr_sender:
        gap_secs = (curr_time - prev_time).total_seconds()
        # only count gaps under 3 hours (ignore overnight silences)
        if 0 < gap_secs < 10800:
            response_gaps[curr_sender].append(gap_secs)
    prev_sender = curr_sender
    prev_time = curr_time
# calculate average response time per person in minutes
avg_response = {}
for name, gaps in response_gaps.items():
    if len(gaps) > 0:
        avg_secs = sum(gaps) / len(gaps)
        avg_response[name] = avg_secs / 60   # convert to minutes
    else:
        avg_response[name] = 99999   # very high if no data
# find fastest and slowest replier
fastest = min(avg_response, key=lambda x: avg_response[x])
slowest = max(avg_response, key=lambda x: avg_response[x])
# format time nicely: if < 60 mins show minutes, else show hours
def format_time(minutes):
    if minutes < 60:
        return f'{round(minutes, 1)} minutes'
    else:
        return f'{round(minutes / 60, 1)} hours'
print()
print('  RESPONSE PATTERNS')
print(f'  Fastest replier : {fastest} (avg {format_time(avg_response[fastest])})')

def spammer_score(name):
